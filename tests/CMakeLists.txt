set(subdirs "")

if(HPX_BUILD_TESTS_ABI)
  set(subdirs
    ${subdirs} abi)
endif()

if(HPX_BUILD_TESTS_BENCHMARKS)
  set(subdirs
    ${subdirs} performance)
endif()

if(HPX_BUILD_TESTS_CORRECTNESS)
  set(subdirs
    ${subdirs} correctness)
endif()

if(HPX_BUILD_TESTS_REGRESSIONS)
  set(subdirs
    ${subdirs} regressions)
endif()

if(HPX_BUILD_TESTS_UNIT)
  set(subdirs
    ${subdirs} unit)
endif()

foreach(subdir ${subdirs})
  add_hpx_pseudo_target(tests.${subdir})
  add_subdirectory(${subdir})
  add_hpx_pseudo_dependencies(tests tests.${subdir})
endforeach()
