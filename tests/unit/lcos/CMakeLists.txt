set(tests
    barrier
    dataflow
    eager_future
    lazy_future
    local_barrier
    local_mutex_lock
    local_mutex_lock_raii
    local_mutex_lock_sanity
    local_mutex_try_lock
    local_mutex_try_lock_raii
    local_mutex_try_lock_sanity
    signalling_promise
   )

set(dataflow_FLAGS DEPENDENCIES dataflow_component)

foreach(test ${tests})
  set(sources
      ${test}.cpp)

  source_group("Source Files" FILES ${sources})

  # add example executable
  add_hpx_executable(${test}_test
                     SOURCES ${sources}
                     ${${test}_FLAGS}
                     FOLDER "Tests/Unit/LCOs")

  # add a custom target for this example
  add_hpx_pseudo_target(tests.unit.lcos.${test})

  # make pseudo-targets depend on master pseudo-target
  add_hpx_pseudo_dependencies(tests.unit.lcos
                              tests.unit.lcos.${test})

  # add dependencies to pseudo-target
  add_hpx_pseudo_dependencies(tests.unit.lcos.${test}
                              ${test}_test_exe)
endforeach()

set_target_properties(dataflow_test_exe
    PROPERTIES COMPILE_FLAGS
        "-DHPX_ACTION_ARGUMENT_LIMIT=10 -DHPX_COMPONENT_CREATE_ARG_MAX=10")
